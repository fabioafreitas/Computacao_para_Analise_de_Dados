flu <- read.csv("flu.csv", header = T)
flu <- flu$age
hist(flu, col = 'grey', probability = T)
densityFlu <- density(flu)
lines(densityFlu)

n <- 200
TamMedia <- 35
xbar <- rep(NA, n)

  MinhaAmostra <- sample(flu, size = TamMedia)
  xbar[i] <- mean(MinhaAmostra)
}
